// ==UserScript==
// @name         Bizgo Thermal Label Printer (40x30mm)
// @namespace    http://tampermonkey.net/
// @version      1.2
// @match        *://mz.bizgo.com/*
// @match        *://*.bizgo.com/*
// @grant        none
// ==/UserScript==
// Cek domain
const _domainCheck = () => {
    if (!window.location.hostname.includes('bizgo.com')) {
        throw new Error('Script execution halted: Unauthorized domain.');
    }
};
_domainCheck();

class DataExtractor {
    static extractAllTabs() {
        const products = [];
        // Cari semua input Nomor SO yang ada di DOM (termasuk di tab yang lagi hidden)
        const soOrderInputs = document.querySelectorAll('input[id^="viewSO-orderNumber"]');

        if (soOrderInputs.length === 0) {
            console.warn("Tidak ada input SO yang ditemukan.");
            return products;
        }

        // Looping setiap tab SO yang lagi kebuka
        soOrderInputs.forEach((soInput) => {
            // Dapet nomor SO-nya
            const soNumber = (soInput.getAttribute('title')) || (soInput.value) || '-';

            // Dapet angka unik (timestamp) dari tab ini buat nyari tabel dan data lainnya
            const suffix = soInput.id.replace('viewSO-orderNumber', '');

            // Cari Customer khusus di tab ini
            const customerInput = document.getElementById('viewSO-customerName' + suffix);
            const customerName = customerInput ? ((customerInput.getAttribute('title')) || (customerInput.value) || '-') : '-';

            // Cari Tanggal di dalem kontainer tab yang sama
            let container = soInput.parentElement;
            let saleDateInput = null;
            while(container && container !== document.body) {
                saleDateInput = container.querySelector('input.salerecdate');
                if (saleDateInput) break;
                container = container.parentElement;
            }
            const saleDate = saleDateInput ? ((saleDateInput.getAttribute('title')) || (saleDateInput.value) || '-') : '-';

            // Eksekusi tarikan data jqxGrid dari memori khusus tab ini
            const gridId = 'viewSaleOrder_grid' + suffix;
            const gridElement = document.getElementById(gridId);

            if (gridElement && window.$) {
                const rows = window.$('#' + gridId).jqxGrid('getrows');
                if (rows && rows.length > 0) {
                    rows.forEach((row) => {
                        const lengthVal = row.extent !== undefined ? row.extent : '0';
                        const widthVal = row.width !== undefined ? row.width : '0';
                        const heightVal = row.height !== undefined ? row.height : '0';
                        const remarkVal = row.remark || '-';
                        const productName = row.name || row.descr || '-';

                        // Parse Quantity, default 1 kalau error atau kosong
                        const rawQty = parseInt(row.qty, 10);
                        const qtyVal = isNaN(rawQty) || rawQty < 1 ? 1 : rawQty;

                        // Looping push produk berdasarkan qty-nya
                        for (let i = 0; i < qtyVal; i++) {
                            products.push({
                                soNumber: soNumber,
                                saleDate: saleDate,
                                customerName: customerName,
                                productName: productName,
                                length: lengthVal,
                                width: widthVal,
                                height: heightVal,
                                remark: remarkVal
                            });
                        }
                    });
                }
            }
        });

        return products;
    }
}

class PrintService {
    static printLabels(dataList) {
        if (dataList.length === 0) {
            alert("Data produk kosong! Pastikan ada produk di tab SO yang terbuka.");
            return;
        }

        console.log(`Mempersiapkan print untuk total ${dataList.length} label dari semua tab...`);

        const printWindow = window.open('', '_blank', 'width=400,height=400');
        if (!printWindow) {
            alert("Tolong izinkan pop-up browser untuk web ini biar bisa nge-print.");
            return;
        }
        const htmlContent = `
            <!DOCTYPE html>
            <html lang="en">
            <head>
                <meta charset="UTF-8">
                <title>Bizgo Label Print</title>
                <style>
                    @page { size: 40mm 30mm; margin: 0; }
                    body { margin: 0; padding: 1.5mm; font-family: Arial, sans-serif; font-size: 8px; line-height: 1.3; width: 40mm; height: 30mm; box-sizing: border-box; color: #000; }
                    .label-page { width: 100%; height: 100%; page-break-after: always; overflow: hidden; display: flex; flex-direction: column; }
                    .label-page:last-child { page-break-after: avoid; }
                    .line { margin-bottom: 2px; word-wrap: break-word; }
                    .line-nowrap { margin-bottom: 2px; white-space: nowrap; overflow: hidden; text-overflow: ellipsis; }
                    .so-header { font-weight: bold; border-bottom: 1px solid #000; padding-bottom: 1px; margin-bottom: 2px; font-size: 9px; }
                </style>
            </head>
            <body>
                ${dataList.map(data => `
                    <div class="label-page">
                        <div class="line-nowrap so-header">SO: ${data.soNumber}</div>
                        <div class="line-nowrap">Sale date: ${data.saleDate}</div>
                        <div class="line-nowrap">Customer: ${data.customerName}</div>
                        <div class="line">Name: ${data.productName}</div>
                        <div class="line">Size: ${data.length}/${data.width}/${data.height}</div>
                        <div class="line">Remark: ${data.remark}</div>
                    </div>
                `).join('')}
                <script>
                    window.onload = () => {
                        window.print();
                        setTimeout(() => window.close(), 500);
                    };
                </script>
            </body>
            </html>
        `;
        printWindow.document.open();
        printWindow.document.write(htmlContent);
        printWindow.document.close();
    }
}

class UIInjector {
    static injectFloatingButton() {
        const existingBtn = document.getElementById('bizgo-tampermonkey-print');
        if (existingBtn) return;
        const btn = document.createElement('button');
        btn.id = 'bizgo-tampermonkey-print';
        btn.textContent = '🖨️ Print Label V3.1 by Bayu';
        Object.assign(btn.style, {
            position: 'fixed', bottom: '30px', left: '30px', padding: '12px 16px',
            backgroundColor: '#007bff', color: '#fff', border: 'none', borderRadius: '8px',
            cursor: 'grab', zIndex: '99999', boxShadow: '0 4px 6px rgba(0,0,0,0.3)',
            fontWeight: 'bold', fontSize: '14px', userSelect: 'none'
        });

        let isDragging = false;
        let isMoved = false;
        let startX, startY;

        const savedPos = localStorage.getItem('bizgoBtnPos');
        if (savedPos) {
            const pos = JSON.parse(savedPos);
            btn.style.transform = `translate(${pos.x}px, ${pos.y}px)`;
            btn.dataset.x = pos.x;
            btn.dataset.y = pos.y;
        } else {
            btn.dataset.x = 0;
            btn.dataset.y = 0;
        }

        btn.addEventListener('mousedown', (e) => {
            isDragging = true;
            isMoved = false;
            btn.style.cursor = 'grabbing';
            startX = e.clientX - parseFloat(btn.dataset.x);
            startY = e.clientY - parseFloat(btn.dataset.y);
        });

        document.addEventListener('mousemove', (e) => {
            if (!isDragging) return;
            isMoved = true;
            const newX = e.clientX - startX;
            const newY = e.clientY - startY;
            btn.style.transform = `translate(${newX}px, ${newY}px)`;
            btn.dataset.x = newX;
            btn.dataset.y = newY;
        });

        document.addEventListener('mouseup', () => {
            if (isDragging) {
                isDragging = false;
                btn.style.cursor = 'grab';
                localStorage.setItem('bizgoBtnPos', JSON.stringify({ x: btn.dataset.x, y: btn.dataset.y }));
            }
        });

        btn.addEventListener('click', (e) => {
            e.preventDefault();
            if (isMoved) return; // Mencegah kepencet pas lagi nge-drag

            // Eksekusi pencetakan
            const extractedData = DataExtractor.extractAllTabs();
            PrintService.printLabels(extractedData);
        });
        document.body.appendChild(btn);
    }
}

const initPrintInjection = () => {
    setInterval(() => {
        UIInjector.injectFloatingButton();
    }, 3000);
};

initPrintInjection();
